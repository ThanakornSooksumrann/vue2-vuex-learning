Vuex — State Management

ปัญหา: ถ้า 2 component ที่ไม่ใช่ parent-child กัน (เช่นอยู่คนละหน้า) ต้องใช้ข้อมูลร่วมกัน (เช่น ข้อมูล user ที่ login, ข้อมูลตะกร้าสินค้า) การส่งผ่าน props/emit จะยุ่งยากมาก Vuex คือ "คลังข้อมูลกลาง" ที่ทุก component เข้าถึงได้โดยตรง โปรเจกต์นี้ใช้ Vuex 3 (คู่กับ Vue 2) ร่วมกับ library เสริมชื่อ vuex-pathify ที่ทำให้เขียนสั้นลง (แนะนำท้ายบทนี้ รายละเอียดเต็มบทที่ 07)

1. 4 ส่วนหลักของ Vuex

ข้อมูล

State

ตัวแปรกลางที่ทุก component อ่านได้ เหมือน data ของแอปทั้งแอป

อ่านแบบคำนวณ

Getters

เหมือน computed แต่ใช้ร่วมกันได้ทั้งแอป (เช่น กรอง/รวมยอด state)

แก้ข้อมูล (sync)

Mutations

ฟังก์ชันเดียวที่อนุญาตให้แก้ state ได้ตรง ๆ ต้องเป็น synchronous เท่านั้น

Logic + async

Actions

เรียก API ได้ (async) แล้วค่อย commit mutation เพื่อเปลี่ยน state จริง

ทำไมต้องแยก mutation กับ action?
กฎ: Component ห้าม commit mutation ตรง ๆ ที่มี async ปนอยู่ — component จะเรียก dispatch(action) เท่านั้น action ไปเรียก API (รอผลลัพธ์) แล้วค่อย commit(mutation) เพื่อเปลี่ยนค่าจริงแบบ synchronous กฎนี้ทำให้ debug ง่าย: ทุกการเปลี่ยน state จะสืบย้อนได้ว่ามาจาก mutation ตัวไหน 
import Vue from 'vue';
import Vuex from 'vuex';
Vue.use(Vuex);

const store = new Vuex.Store({
  state: {
    count: 0
  },
  getters: {
    doubleCount(state) {
      return state.count * 2;
    }
  },
  mutations: {
    INCREMENT(state, amount) {
      state.count += amount;   // แก้ state ตรง ๆ ได้เฉพาะใน mutation เท่านั้น
    }
  },
  actions: {
    incrementAsync({ commit }, amount) {
      return new Promise(resolve => {
        setTimeout(() => {              // จำลองการเรียก API
          commit('INCREMENT', amount);
          resolve();
        }, 500);
      });
    }
  }
});

// ใช้งานใน component
this.$store.state.count            // อ่านค่าตรง ๆ
this.$store.getters.doubleCount    // อ่าน getter
this.$store.commit('INCREMENT', 1) // เรียก mutation ตรง ๆ (ใช้กับ sync เท่านั้น)
this.$store.dispatch('incrementAsync', 5) // เรียก action (รองรับ async)

2. Modules — แยกร้านค้าออกจากกันเมื่อแอปใหญ่ขึ้น

แอปใหญ่มี state เยอะมาก ถ้าเก็บไว้ก้อนเดียวจะดูแลยาก จึงแยกเป็น module ย่อยตามฟีเจอร์/หน้าจอ และควรเปิด namespaced: true เสมอ เพื่อกันชื่อ action/mutation ชนกันข้าม module 

// store/modules/user.js
export default {
  namespaced: true,           // สำคัญมาก! ถ้าไม่เปิด ชื่อ action จะไปปนกับ module อื่น
  state: () => ({
    profile: null
  }),
  getters: {
    displayName: (state) => state.profile ? state.profile.name : 'Guest'
  },
  mutations: {
    SET_PROFILE(state, profile) { state.profile = profile; }
  },
  actions: {
    async fetchProfile({ commit }, userId) {
      const res = await api.get(`/users/${userId}`);
      commit('SET_PROFILE', res.data);
    }
  }
};

// store/index.js
import user from './modules/user';
export default new Vuex.Store({
  modules: { user }
});

// ใช้งานใน component (ต้องใส่ path ของ module นำหน้าเสมอ)
this.$store.state.user.profile
this.$store.getters['user/displayName']
this.$store.dispatch('user/fetchProfile', 123)
แบบที่โปรเจกต์นี้ใช้จริง
โปรเจกต์นี้มี Vuex module เป็นหลายร้อยตัว (แทบจะ 1 module ต่อ 1 หน้าจอ/component ย่อย) ทุก module เปิด namespaced: true เสมอ และไม่ได้เขียน import module ทีละไฟล์เอง แต่ใช้ Webpack require.context ไล่โหลดไฟล์ทุกไฟล์ในโฟลเดอร์ store/shared และ store/views อัตโนมัติ แล้วตั้งชื่อ module จาก path ของไฟล์ เช่น src/app/store/shared/Layout/Stores/Main.js → module ชื่อ Layout_Main (รายละเอียดเต็ม ๆ บทที่ 07)

3. ตัวอย่างจริง: module ที่คุยกับ API จริง (loading / error / CRUD)

ของจริงในงานแทบทุกวันไม่ใช่ตัวนับเลข แต่เป็น "โหลดลิสต์ข้อมูลจาก API, เพิ่ม/ลบรายการ, มี loading ตอนรอ และมี error ตอนพัง" ตัวอย่างนี้คือ Vuex module ของ "รายการสิ่งที่ต้องทำ" (todos) แบบที่ต้องเขียนใช้งานจริง ครบทั้ง loading state, error handling, และ action ที่ต้อง await ผลลัพธ์ก่อนค่อยไปทำ step ถัดไป 

// store/modules/todos.js
export default {
  namespaced: true,
  state: () => ({
    items: [],
    loading: false,
    error: ""
  }),
  getters: {
    doneCount: (state) => state.items.filter(t => t.done).length
  },
  mutations: {
    SET_LOADING(state, val) { state.loading = val; },
    SET_ERROR(state, msg) { state.error = msg; },
    SET_ITEMS(state, items) { state.items = items; },
    ADD_ITEM(state, item) { state.items.push(item); },
    REMOVE_ITEM(state, id) { state.items = state.items.filter(t => t.id !== id); },
    TOGGLE_ITEM(state, id) {
      const item = state.items.find(t => t.id === id);
      if (item) item.done = !item.done;
    }
  },
  actions: {
    // โหลดข้อมูลจาก API — เรียกตอน created() ของหน้าจอ
    async fetchTodos({ commit }) {
      commit('SET_LOADING', true);
      commit('SET_ERROR', '');
      try {
        const res = await axios.get('/api/todos');
        commit('SET_ITEMS', res.data);
      } catch (err) {
        commit('SET_ERROR', 'โหลดรายการไม่สำเร็จ');
      } finally {
        commit('SET_LOADING', false);
      }
    },
    // เพิ่มรายการ — เรียก API แล้วค่อย commit ถ้าสำเร็จเท่านั้น (ไม่ commit มั่วก่อนรู้ผล)
    async addTodo({ commit }, text) {
      if (!text.trim()) {
        commit('SET_ERROR', 'กรุณากรอกข้อความก่อนเพิ่มรายการ');
        return false;
      }
      try {
        const res = await axios.post('/api/todos', { text });
        commit('ADD_ITEM', res.data);   // ใช้ข้อมูลที่ server ส่งกลับมา (มี id จริงจาก DB)
        return true;
      } catch (err) {
        commit('SET_ERROR', 'เพิ่มรายการไม่สำเร็จ');
        return false;
      }
    },
    async removeTodo({ commit }, id) {
      await axios.delete(`/api/todos/${id}`);
      commit('REMOVE_ITEM', id);
    }
  }
};

ใช้งานจาก component จริง — สังเกตว่า component ไม่ยุ่งกับ axios โดยตรงเลย รู้แค่ว่า "dispatch action แล้วรอผล" ส่วนจะเรียก API ยังไง, error format แบบไหน ปล่อยให้ module จัดการทั้งหมด:

export default {
  computed: {
    ...mapState('todos', ['items', 'loading', 'error']),
    ...mapGetters('todos', ['doneCount'])
  },
  methods: {
    ...mapActions('todos', ['fetchTodos', 'addTodo', 'removeTodo']),
    async onAddClick() {
      const ok = await this.addTodo(this.newText);   // action คืนค่า true/false บอกผลลัพธ์
      if (ok) this.newText = "";                       // เคลียร์ช่อง input เฉพาะตอนสำเร็จ
    }
  },
  created() {
    this.fetchTodos();   // โหลดข้อมูลตั้งแต่เปิดหน้า (ดูบทที่ 02 เรื่อง lifecycle)
  }
};

4. mapState / mapGetters / mapMutations / mapActions — ทางลัดใน component

import { mapState, mapGetters, mapActions } from 'vuex';

export default {
  computed: {
    ...mapState('user', ['profile']),          // = this.$store.state.user.profile
    ...mapGetters('user', ['displayName'])     // = this.$store.getters['user/displayName']
  },
  methods: {
    ...mapActions('user', ['fetchProfile'])    // this.fetchProfile(123) แทน dispatch เต็ม ๆ
  },
  created() {
    this.fetchProfile(123);
  }
};
โปรเจกต์นี้ไม่ได้ใช้ mapState/mapActions แบบมาตรฐาน — ใช้ vuex-pathify แทน
vuex-pathify คือ library ที่ทำให้เขียนสั้นลงไปอีก โดยจะgenerate mutation/action พื้นฐานให้อัตโนมัติ (get/set ของทุก property) และมี helper sync() ที่รวม getter+setter ให้ในบรรทัดเดียว: 
import { sync } from 'vuex-pathify';
export default {
  computed: {
    ...sync('user/profile'),          // อ่าน + เขียนได้เลย เหมือน v-model ตรง ๆ กับ store
    ...sync('user/*')                 // sync ทุก property ใน module เดียว
  },
  methods: {
    save() {
      this.$store.set('user/fetchProfile!', 123);  // สังเกต '!' ต่อท้าย = เรียก action ไม่ใช่ set ค่าตรง ๆ
      let value = this.$store.get('user/profile');  // อ่านค่าตรง ๆ แบบ path string
    }
  }
};
เครื่องหมาย ! ต่อท้ายชื่อคือ convention ของ pathify ที่บอกว่า "นี่คือ action ให้ dispatch นะ ไม่ใช่ mutation ให้ set ค่าตรง ๆ" จำ syntax นี้ให้ขึ้นใจเพราะจะเจอในทุกไฟล์ของโปรเจกต์นี้ — รายละเอียดเต็มอยู่บทที่ 07

สรุปสิ่งที่ต้องจำ

  • State = ข้อมูลกลาง, Getters = computed กลาง, Mutations = จุดเดียวที่แก้ state ได้ (sync เท่านั้น), Actions = logic/async แล้วค่อย commit
  • Component เรียก dispatch(action) ไม่ commit mutation ตรง ๆ เวลามี async
  • ใช้ modules + namespaced: true เมื่อแอปใหญ่ขึ้น เพื่อกันชื่อชนกัน
  • โปรเจกต์นี้ใช้ vuex-pathify เป็นตัวช่วยเขียนสั้นลง ผ่าน sync(), $store.get(), $store.set() และ action ต้องมี ! ต่อท้าย

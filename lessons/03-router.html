Vue Router

Vue Router ทำให้แอปมีหลาย "หน้า" ได้โดยไม่ต้อง reload browser (Single Page Application) — โปรเจกต์นี้ใช้ Vue Router 3 (คู่กับ Vue 2) โหมด history แต่ที่พิเศษคือ route ทั้งหมดถูกสร้างอัตโนมัติจากโครงสร้างโฟลเดอร์ views ไม่ได้เขียน route ทีละบรรทัดเอง (รายละเอียดเต็ม ๆ อยู่บทที่ 07)

1. concept หลัก 3 อย่าง

ที่เดียว ต่อ 1 แอป

routes

รายการ mapping ระหว่าง path (เช่น /customer) กับ component ที่จะแสดง

วางในหน้า layout

<router-view>

"ช่อง" ที่ component ของ route ปัจจุบันจะไป render ใส่

ใช้แทน <a>

<router-link>

ลิงก์เปลี่ยนหน้าโดยไม่ reload browser (SPA navigation)

// router/index.js
import Vue from 'vue';
import VueRouter from 'vue-router';
import Home from './views/Home.vue';
import Customer from './views/Customer.vue';

Vue.use(VueRouter);

const routes = [
  { path: '/', component: Home },
  { path: '/customer', component: Customer }
];

const router = new VueRouter({
  mode: 'history',   // URL สวย ไม่มี # (ต้องตั้งค่า server ให้ fallback มาที่ index.html)
  routes
});

export default router;

// main.js
new Vue({
  router,             // ผูก router เข้ากับ Vue instance
  render: h => h(App)
}).$mount('#app');

// App.vue
<template>
  <div>
    <router-link to="/">หน้าแรก</router-link>
    <router-link to="/customer">ลูกค้า</router-link>
    <router-view />   <!-- component ของ route ปัจจุบันจะมาโผล่ตรงนี้ -->
  </div>
</template>
ตัวอย่างสด (ใช้โหมด hash เพื่อให้รันในไฟล์ static ได้โดยไม่ต้องมี server) ● LIVE
หน้าแรก | ลูกค้า #1001 | ลูกค้า #2002 (แท็บ history)

2. Dynamic params และ Query string

รูปแบบ pathอ่านค่าใน component
Params (ส่วนของ path) /customer/:id → /customer/1001 this.$route.params.id → "1001"
Query (ต่อท้ายด้วย ?) /customer/1001?tab=history this.$route.query.tab → "history"
แบบที่โปรเจกต์นี้ใช้จริง
โปรเจกต์นี้แทบไม่ใช้ dynamic params (:id) เลย แต่จะส่งค่าผ่าน query string เป็นหลัก เช่น /Finance/InquiryCashReceiptUnknownContract/Index?mode=Inquiry&modeFlag=1 แล้วรับค่าผ่าน props mode, modeFlag, modePage ที่ router ผูกให้อัตโนมัติทุก route (ดูบทที่ 07)

3. Navigation Guard — ดักก่อน/หลังเปลี่ยนหน้า

router.beforeEach((to, from, next) => {
  // to   = route ที่กำลังจะไป
  // from = route ปัจจุบัน (ก่อนเปลี่ยน)
  // next() = อนุญาตให้ไปต่อ, next(false) = ยกเลิก, next('/login') = เปลี่ยนเส้นทางแทน
  if (to.meta.requiresAuth && !isLoggedIn()) {
    next('/login');
  } else {
    next();
  }
});

router.afterEach((to, from) => {
  // ทำงานหลังเปลี่ยนหน้าสำเร็จแล้ว เช่น ส่ง analytics, ตั้งชื่อ title
  document.title = to.meta.title || 'My App';
});
แบบที่โปรเจกต์นี้ใช้จริง src/app/router/index.js
router.beforeEach ใช้ปิดการเชื่อมต่อ SignalR hub บางประเภทก่อนเปลี่ยนหน้า
router.afterEach ใช้เคลียร์ temp data, เช็คว่ากำลังไปหน้า sign-in หรือเปล่า (ถ้าใช่ให้ reset store ทุก module), และเรียก setTitle() เพื่ออัปเดตชื่อหัวข้อ/breadcrumb ด้านบนจาก MenuData

4. Lazy loading — โหลด component เฉพาะตอนเข้าหน้านั้นจริง ๆ

// แทนที่จะ import ตรง ๆ (โหลดพร้อมกันหมดตั้งแต่แรก)
import Customer from './views/Customer.vue';

// ใช้ dynamic import แทน (โหลดเป็นไฟล์แยก โหลดเมื่อเข้า route นี้เท่านั้น)
const routes = [
  { path: '/customer', component: () => import('./views/Customer.vue') }
];

สำคัญมากสำหรับแอปใหญ่แบบนี้ที่มีหน้าจอเป็นพันหน้า — ถ้าโหลดทุกหน้าตั้งแต่แรกจะทำให้หน้าเว็บโหลดช้ามาก โปรเจกต์นี้ lazy-load ทุก route โดยอัตโนมัติ อยู่แล้ว (ดูบทที่ 07 ว่าทำยังไง)

5. Programmatic navigation — เปลี่ยนหน้าด้วยโค้ด

this.$router.push('/customer/1001');
this.$router.push({ path: '/customer', query: { tab: 'history' } });
this.$router.back();          // ย้อนกลับหน้าก่อนหน้า (เหมือนกด back)
this.$router.replace('/login'); // เปลี่ยนหน้าโดยไม่เก็บ history (กด back แล้วไม่ย้อนมาหน้านี้)

สรุปสิ่งที่ต้องจำ

  • routes = แมพ path กับ component, <router-view> = ช่องแสดงผล, <router-link> = ลิงก์เปลี่ยนหน้า
  • อ่านค่าจาก URL ผ่าน this.$route.params (path) และ this.$route.query (query string)
  • beforeEach/afterEach ใช้ดัก logic ก่อน-หลังเปลี่ยนหน้า (auth guard, set title)
  • ใช้ () => import(...) เพื่อ lazy-load component ของแต่ละหน้า
  • โปรเจกต์นี้ auto-generate route จากโฟลเดอร์ views และส่ง mode/modeFlag/modePage ผ่าน query string เป็น convention หลัก

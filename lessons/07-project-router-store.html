Router & Vuex ของโปรเจกต์นี้ (ตัวจริงที่ทีมทำไว้แล้ว)

นี่คือบทที่สำคัญที่สุดในการ "เริ่มงานได้จริง" เพราะโปรเจกต์นี้ไม่ได้เขียน route หรือ vuex module มือทีละไฟล์ แต่ใช้ Webpack require.context สแกนโฟลเดอร์แล้วสร้างให้อัตโนมัติ — ถ้าไม่รู้กลไกนี้จะงงมากว่า "ทำไมหน้าจอนี้เข้าได้เองทั้งที่ไม่เห็นใครไปเพิ่ม route ที่ไหนเลย"

1. Router auto-generation — route มาจากชื่อไฟล์ .vue ในโฟลเดอร์ views

ไฟล์จริง: src/app/router/index.js

let routes = [];

let importRouterScreens = (req) => {
  let keys = req.keys();
  keys.forEach(function (key) {
    let match = key.match(/([^\/]+)(?=\.\w+$)/) || [];
    if (match.length > 0) {
      let name = match[0];
      let fileName = key.split("/").pop();
      let path = `${key.replace("./","/").replace(fileName,name)}`;

      routes.push({
        path: key.replace("./","/").replace(fileName,name),
        component: () => import("../views" + path),   // lazy-load อัตโนมัติ
        props: route => ({
          mode: route.query.mode,
          modeFlag: route.query.modeFlag,
          modePage: route.query.modePage
        })
      });
    }
  });
};

// สแกนทุกไฟล์ .vue ใต้ src/app/views/ แบบ recursive
importRouterScreens(require.context("../views", true, /^(?=.*.vue$)/, 'weak'));
แปลว่าอะไร: กฎทองของโปรเจกต์นี้
ตำแหน่งไฟล์ .vue ใน src/app/views/ = path ของ route โดยอัตโนมัติ ไม่ต้องไปเพิ่มไฟล์ router เอง

ตัวอย่าง: ไฟล์ src/app/views/Finance/InquiryCashReceiptUnknownContract/Index.vue จะกลายเป็น route path /Finance/InquiryCashReceiptUnknownContract/Index โดยอัตโนมัติทันทีที่สร้างไฟล์นี้ขึ้นมา

ทุก route ที่สร้างแบบนี้จะได้ props อัตโนมัติ 3 ตัว จาก query string เสมอ:

Propมาจาก queryใช้ทำอะไรทั่วไป
mode?mode=Inquiryบอกว่าหน้านี้เปิดมาแบบไหน เช่น Inquiry (ดูอย่างเดียว), Edit, Approve
modeFlag?modeFlag=1flag เพิ่มเติมแยกย่อยพฤติกรรมภายใน mode เดียวกัน
modePage?modePage=xxxบอกบริบทของหน้าที่เปิดมา (เช่นเปิดจากเมนูไหน)
 
// ในหน้าจอ (Index.vue) รับค่าผ่าน props ได้เลย ไม่ต้องอ่าน $route.query เอง
export default {
  props: {
    mode: { type: String, default() { return ""; } }
  },
  computed: {
    isInquiryMode() { return this.mode === 'Inquiry'; }
  }
};

2. Router Guard ของโปรเจกต์ (ของจริง)

// beforeEach: ปิด SignalR hub บางประเภทก่อนเปลี่ยนหน้า (กันหลุด/ค้าง)
router.beforeEach((to, from, next) => {
  const vm = (router.app || {}).$root;
  const hubConnections = vm._getHubConnections();
  const promises = hubConnections
    .filter(hub => hub.hubType === 1)
    .map(hub => hub.dispose());
  Promise.all(promises).then(() => next());
});

// afterEach: เคลียร์ค่าตกค้าง + reset store เมื่อไปหน้า sign-in + อัปเดต title
router.afterEach((to, from) => {
  const path = to.path.toLowerCase();
  if (path.indexOf("signin") > -1) {
    store.resetAllModules();   // reset ทุก module ยกเว้น Layout_* กันข้อมูล user เก่าค้าง
  } else {
    store.set("Layout_Main/isSetTitle", false).then(() => {
      app.setTitle();          // อัปเดตชื่อหัวข้อ/breadcrumb จาก MenuData ตาม path ปัจจุบัน
    });
  }
});

3. Vuex module auto-registration — เหมือน router เป๊ะ ๆ

ไฟล์จริง: src/app/store/index.js

let modules = {};
let importKeys = (route, req) => {
  req.keys().forEach((key) => {
    // แปลง path ไฟล์ให้เป็นชื่อ module เช่น
    // "./Layout/Stores/Main.js" -> "Layout_Main"
    let moduleName = key
      .replace("/Stores", "")
      .replace(".js", "")
      .replace(/\.\//gi, "")
      .replace(/\//gi, "_");

    modules[moduleName] = req(key).default;
  });
};

// สแกนทุกไฟล์ .js ที่อยู่ใต้โฟลเดอร์ Stores/ ทั้งใน shared และ views
importKeys("", require.context("faco.core@app/store/shared/", true, /\/Stores\/.*\.js/i));
importKeys("", require.context("faco.core@app/store/views/", true, /\/Stores\/.*\.js/i));

const store = new Vuex.Store({
  modules,
  plugins: [pathify.plugin]
});
กฎทองข้อที่ 2: ชื่อ module = path ของไฟล์ store
ไฟล์ store จริงชื่อ module ที่ได้
store/shared/Layout/Stores/Main.jsLayout_Main
store/shared/Layout/Stores/MasterLayout.jsLayout_MasterLayout
store/views/Demo/Stores/Demo2.jsDemo_Demo2
store/views/Finance/.../Stores/InquiryCashReceiptUnknownContractDetail.jsFinance_InquiryCashReceiptUnknownContract_InquiryCashReceiptUnknownContractDetail

ดังนั้นเวลาสร้าง store module ใหม่ ต้องวางไฟล์ในตำแหน่งที่ถูกต้อง เพราะชื่อ module จะถูกคำนวณจาก path เสมอ ไม่ได้ตั้งชื่อเองอิสระ

4. Store module pattern จริงของโปรเจกต์ (vuex-pathify make)

ทุก module ใช้ make.getters/mutations/actions จาก vuex-pathify เพื่อ auto-generate getter/mutation/action ของทุก state property ให้ฟรี ไม่ต้องเขียนเอง

// src/app/store/shared/Layout/Stores/Main.js (ของจริง)
import { make } from 'vuex-pathify';
import axios from "faco.core@repository";

function state() {
  return {
    completedAuth: false,
    isLoginPage: false,
    tempFiles: []
  };
}

const getters = {
  ...make.getters(state)     // ได้ getter "completedAuth", "isLoginPage" ฯลฯ ให้ฟรี
};

const mutations = {
  ...make.mutations(state),  // ได้ mutation "SET_COMPLETED_AUTH" ฯลฯ ให้ฟรี (ตั้งชื่อ SET_ ใหญ่ตัวพิมพ์)
  clearTempFiles(state) {    // เขียน mutation เองเพิ่มได้ตามต้องการ
    state.tempFiles.splice(0);
  }
};

const actions = {
  ...make.actions(state),    // ได้ action "completedAuth" ฯลฯ ให้ฟรี (เรียกผ่าน pathify "!" )
  getSystemDate({ commit }) {
    return new Promise(resolve => {
      axios.systemConfigApi.get(`System/SystemDate`).then(response => {
        commit("tempSysDate", response.data.content);
        resolve(true);
      }).catch(() => resolve(false));
    });
  }
};

export default { namespaced: true, state, getters, actions, mutations };

5. วิธีใช้ store จาก component ด้วย pathify (ของจริงที่เจอทุกไฟล์)

import { sync, get } from "vuex-pathify";

var pathStore = {
  detail: "Finance_InquiryCashReceiptUnknownContract_InquiryCashReceiptUnknownContractDetail"
};

export default {
  computed: {
    // sync ทุก property ของ module นี้ในทีเดียว — อ่าน/เขียนได้เหมือน data ปกติ
    ...sync(`${pathStore.detail}/*`)
    // ตอนนี้จะมี this.dataInGrid, this.columns ฯลฯ ให้ใช้ตรง ๆ ถ้า state module มี property เหล่านั้น
  },
  methods: {
    onSearch() {
      let filter = this.$refs.filterForm.onSearchFilter();

      // เรียก action ต้องมี "!" ต่อท้ายชื่อเสมอ — บอก pathify ว่านี่คือ action ไม่ใช่ set ค่า state ตรง ๆ
      this.$store.set(`${pathStore.detail}/getDataInquiryCashReceiptUnknownContract!`, filter)
        .then((result) => {
          if (this.dataInGrid.length < 1) {
            this.$refs.alertMessage.open("คำเตือน", "ไม่พบข้อมูล", 400);
          }
        });
    }
  },
  mounted() {
    // ตั้งชื่อ title ของหน้าจอ ทุกหน้าทำแบบนี้เหมือนกันหมด
    this.$store.set("Layout_MasterLayout/setTitle", {
      titleName: this.$t("FF-LB-003859"),
      subTitleName: `${this.$t('FF-LB-003846')} > ${this.$t('FF-LB-003668')}`
    });
  }
};
Syntaxความหมาย
...sync('Module/*')ผูก state ทุกตัวใน module นั้นเป็น computed (get+set) อัตโนมัติ
...sync('Module/field')ผูกเฉพาะ field เดียว
this.$store.get('Module/field')อ่านค่า state แบบ manual (ไม่ผูก reactive computed)
this.$store.set('Module/field', value)set state ตรง ๆ (เรียก mutation ที่ auto-gen ให้)
this.$store.set('Module/actionName!', payload)เครื่องหมาย ! ต่อท้าย = เรียก action (รองรับ async, return Promise)

store.externalRegister("SomeModuleName", moduleDefinition);

สรุปสิ่งที่ต้องจำ

  • สร้างไฟล์ .vue ใต้ src/app/views/... → ได้ route ให้อัตโนมัติ (path = ตำแหน่งไฟล์)
  • ทุก route ได้ props mode, modeFlag, modePage จาก query string ให้ฟรี
  • สร้างไฟล์ store ใต้ store/shared/.../Stores/ หรือ store/views/.../Stores/ → ได้ Vuex module ให้อัตโนมัติ ชื่อ module มาจาก path
  • ใช้ make.getters/mutations/actions เพื่อไม่ต้องเขียน getter/mutation พื้นฐานเอง
  • ใช้ sync(), $store.get(), $store.set() เสมอแทน mapState/mapActions แบบมาตรฐาน และอย่าลืม ! ต่อท้ายเวลาเรียก action
